import os
import re

import requests
from pypdf import PdfReader
import docx2txt

# --- 1. HÀM ĐỌC TEXT TỪ FILE WORD (.DOCX) ---
def read_word_file(file_path):
    print(f"-> Đang trích xuất chữ từ file Word: {file_path}")
    text = docx2txt.process(file_path)
    return text

# --- 2. HÀM ĐỌC TEXT TỪ FILE PDF (.PDF) ---
def read_pdf_file(file_path):
    print(f"-> Đang trích xuất chữ từ file PDF: {file_path}")
    reader = PdfReader(file_path)
    full_text = ""
    for page in reader.pages:
        page_text = page.extract_text()
        if page_text:
            full_text += page_text + "\n"
    return full_text

# --- 3. THUẬT TOÁN CHIA NHỎ VĂN BẢN (SMART CHUNKING) ---
def smart_text_chunking(text, max_chars=2000):
    # Chuẩn hóa khoảng trắng
    text = re.sub(r'\n+', '\n', text).strip()
    chunks = []
    current_chunk = ""
    
    # Tách văn bản thành các câu dựa trên dấu câu
    sentences = re.split(r'(?<=[.!?\n])\s+', text)
    
    for sentence in sentences:
        if len(current_chunk) + len(sentence) + 1 > max_chars:
            if current_chunk:
                chunks.append(current_chunk.strip())
            current_chunk = sentence
        else:
            if current_chunk:
                current_chunk += " " + sentence
            else:
                current_chunk = sentence
                
    if current_chunk:
        chunks.append(current_chunk.strip())
    return chunks

# --- 4. HÀM GỌI API FPT VÀ TẢI FILE MP3 ---
def text_to_speech_fpt(text, api_key, output_filename):
    url = "https://api.fpt.ai/hcm/v5/vtts"
    headers = {
        "api-key": api_key,
        "speed": "1",
        "voice": "banmai" # Bạn có thể đổi sang 'leminh' (giọng Nam Bắc) hoặc 'lananh' (giọng Nữ Nam)
    }
    
    response = requests.post(url, data=text.encode('utf-8'), headers=headers)
    if response.status_code == 200:
        audio_url = response.json().get("async")
        # Chờ hệ thống tạo file ngầm (đoạn dài cần chờ lâu hơn một chút)
        time.sleep(8) 
        
        audio_response = requests.get(audio_url)
        if audio_response.status_code == 200:
            with open(output_filename, "wb") as f:
                f.write(audio_response.content)
            print(f"   [OK] Đã tạo xong phần: {output_filename}")
        else:
            print(f"   [Lỗi] Không tải được file âm thanh của phần này.")
    else:
        print(f"   [Lỗi API] Mã lỗi: {response.status_code}")

import os
import subprocess

def merge_mp3_files(file_list, output_filename="podcast_hoan_thien.mp3"):
    """
    Hàm tự động nối danh sách các file MP3 nhỏ thành một file Podcast duy nhất bằng FFmpeg.
    """
    # 1. Tạo một file văn bản tạm thời để chứa danh sách các file cần nối
    # FFmpeg yêu cầu danh sách theo định dạng: file 'tên_file.mp3'
    list_filename = "danh_sách_noi_file.txt"
    with open(list_filename, "w", encoding="utf-8") as f:
        for file_path in file_list:
            f.write(f"file '{file_path}'\n")
            
    print(f"\n-> Đang tiến hành ghép {len(file_list)} file nhỏ thành một file duy nhất...")
    
    # 2. Xây dựng lệnh để gọi FFmpeg giống như khi gõ trên CMD
    # -y: Tự động ghi đè nếu file trùng tên đã tồn tại
    # -c copy: Sao chép nguyên bản dữ liệu âm thanh (Xử lý siêu nhanh)
    cmd = [
        "ffmpeg", "-f", "concat", "-safe", "0", 
        "-i", list_filename, "-c", "copy", "-y", output_filename
    ]
    
    try:
        # Chạy lệnh ngầm và ẩn các dòng log thông báo rườm rà của FFmpeg
        subprocess.run(cmd, check=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
        print(f"🎉 THÀNH CÔNG RỰC RỠ! File Podcast hoàn chỉnh đã được tạo: {output_filename}")
        
        # 3. Dọn dẹp: Xóa các file MP3 nhỏ và file text tạm sau khi đã nối xong
        print("-> Đang dọn dẹp các file rác tạm thời...")
        for file_path in file_list:
            if os.path.exists(file_path):
                os.remove(file_path)
        if os.path.exists(list_filename):
            os.remove(list_filename)
            
    except subprocess.CalledProcessError as e:
        print(f"❌ Gặp lỗi khi chạy FFmpeg: {e.stderr.decode('utf-8', errors='ignore')}")
